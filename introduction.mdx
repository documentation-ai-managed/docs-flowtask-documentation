<CodeGroup tabs="JavaScript,Python">
  ````javascript
// Quick start with FlowTask SDK
const flowtask = require('flowtask-sdk');
const client = new flowtask.Client({ apiKey: 'your-api-key' });

// Create a new project
const project = await client.projects.create({
  title: 'My AI Project',
  description: 'Using AI to organize tasks'
});
  ````
  ````python
from flowtask import FlowTask

# Initialize client
client = FlowTask(api_key='your-api-key')

# Create project
project = client.projects.create(
    title='My AI Project',
    description='Using AI to organize tasks'
)
  ````
